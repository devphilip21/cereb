---
title: map
description: Transform each emitted value
---

Applies a projection function to each value emitted by the source Observable.

## Signature

```typescript
function map<T, R>(project: (value: T, index: number) => R): Operator<T, R>
```

## Parameters

| Name | Type | Description |
|------|------|-------------|
| `project` | `(value: T, index: number) => R` | Function to transform each value |

## Usage

```typescript
import { pipe, from } from "@cereb/stream";
import { map } from "@cereb/stream/operators";

const numbers$ = from([1, 2, 3]);

const doubled$ = pipe(
  numbers$,
  map((x) => x * 2)
);

doubled$.subscribe(console.log);
// Output: 2, 4, 6
```

## With Index

```typescript
const indexed$ = pipe(
  from(["a", "b", "c"]),
  map((value, index) => `${index}: ${value}`)
);

indexed$.subscribe(console.log);
// Output: "0: a", "1: b", "2: c"
```

## Real-World Example

```typescript
import { singlePointer } from "@cereb/single-pointer";
import { pipe } from "@cereb/stream";
import { map } from "@cereb/stream/operators";

const pointer$ = singlePointer(element);

const positions$ = pipe(
  pointer$,
  map((signal) => ({ x: signal.x, y: signal.y }))
);
```
