---
title: filter
description: Filter values based on a predicate
---

Emits only values that pass the provided predicate function.

## Signature

```typescript
function filter<T>(predicate: (value: T, index: number) => boolean): Operator<T, T>
```

## Parameters

| Name | Type | Description |
|------|------|-------------|
| `predicate` | `(value: T, index: number) => boolean` | Function to test each value |

## Usage

```typescript
import { pipe, from } from "@cereb/stream";
import { filter } from "@cereb/stream/operators";

const numbers$ = from([1, 2, 3, 4, 5]);

const even$ = pipe(
  numbers$,
  filter((x) => x % 2 === 0)
);

even$.subscribe(console.log);
// Output: 2, 4
```

## Real-World Example

```typescript
import { singlePointer } from "@cereb/single-pointer";
import { pipe } from "@cereb/stream";
import { filter } from "@cereb/stream/operators";

const pointer$ = singlePointer(element);

// Only process "change" phase (movement)
const move$ = pipe(
  pointer$,
  filter((signal) => signal.phase === "change")
);

// Only left button clicks
const leftClick$ = pipe(
  pointer$,
  filter((signal) => signal.button === "left" && signal.phase === "start")
);
```
