Operators for skipping values and removing duplicates.

## skip

Skips the first N values.

```typescript
function skip<T>(count: number): Operator<T, T>
```

### Usage

```typescript
import { pipe, from } from "@cereb/core";
import { skip } from "@cereb/core";

const skipped$ = pipe(
  from([1, 2, 3, 4, 5]),
  skip(2)
);

skipped$.subscribe(console.log);
// Output: 3, 4, 5
```

## skipWhile

Skips values while predicate returns true.

```typescript
function skipWhile<T>(predicate: (value: T) => boolean): Operator<T, T>
```

### Usage

```typescript
import { pipe, from } from "@cereb/core";
import { skipWhile } from "@cereb/core";

const numbers$ = pipe(
  from([1, 2, 3, 4, 3, 2, 1]),
  skipWhile((x) => x < 3)
);

numbers$.subscribe(console.log);
// Output: 3, 4, 3, 2, 1
```

## skipUntil

Skips values until notifier emits.

```typescript
function skipUntil<T>(notifier: Observable<unknown>): Operator<T, T>
```

### Usage

```typescript
import { pipe, fromEvent } from "@cereb/core";
import { skipUntil } from "@cereb/core";

const start$ = fromEvent(startButton, "click");

const stream$ = pipe(
  source$,
  skipUntil(start$)
);
```

## distinctUntilChanged

Emits only when value differs from previous.

```typescript
function distinctUntilChanged<T>(
  compare?: (prev: T, curr: T) => boolean
): Operator<T, T>
```

### Usage

```typescript
import { pipe, from } from "@cereb/core";
import { distinctUntilChanged } from "@cereb/core";

const distinct$ = pipe(
  from([1, 1, 2, 2, 3, 1, 1]),
  distinctUntilChanged()
);

distinct$.subscribe(console.log);
// Output: 1, 2, 3, 1
```

## distinctUntilKeyChanged

Emits only when specified key differs.

```typescript
function distinctUntilKeyChanged<T, K extends keyof T>(key: K): Operator<T, T>
```

### Usage

```typescript
import { pipe } from "@cereb/core";
import { distinctUntilKeyChanged } from "@cereb/core";

const users$ = pipe(
  userUpdates$,
  distinctUntilKeyChanged("id")
);
```
