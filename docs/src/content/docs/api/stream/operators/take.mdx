Operators for limiting emissions.

## take

Emits only the first N values.

```typescript
function take<T>(count: number): Operator<T, T>
```

### Usage

```typescript
import { pipe, from } from "@cereb/core";
import { take } from "@cereb/core";

const limited$ = pipe(
  from([1, 2, 3, 4, 5]),
  take(3)
);

limited$.subscribe(console.log);
// Output: 1, 2, 3
```

## takeWhile

Emits values while predicate returns true.

```typescript
function takeWhile<T>(predicate: (value: T) => boolean): Operator<T, T>
```

### Usage

```typescript
import { pipe, from } from "@cereb/core";
import { takeWhile } from "@cereb/core";

const numbers$ = pipe(
  from([1, 2, 3, 4, 3, 2, 1]),
  takeWhile((x) => x < 4)
);

numbers$.subscribe(console.log);
// Output: 1, 2, 3
```

## takeUntil

Emits values until notifier emits.

```typescript
function takeUntil<T>(notifier: Observable<unknown>): Operator<T, T>
```

### Usage

```typescript
import { pipe, fromEvent } from "@cereb/core";
import { takeUntil } from "@cereb/core";

const stop$ = fromEvent(stopButton, "click");

const stream$ = pipe(
  source$,
  takeUntil(stop$)
);
```

## Real-World Example

```typescript
import { singlePointer } from "@cereb/single-pointer";
import { pipe } from "@cereb/core";
import { takeWhile } from "@cereb/core";

// Stop tracking when pointer leaves the element
const pointer$ = pipe(
  singlePointer(element),
  takeWhile((s) => s.phase !== "cancel")
);
```
