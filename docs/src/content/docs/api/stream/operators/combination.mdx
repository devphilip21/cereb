Operators for combining multiple observables.

## merge

Combines multiple observables into one, emitting values from all.

```typescript
function merge<T>(...sources: Observable<T>[]): Observable<T>
function mergeWith<T>(...sources: Observable<T>[]): Operator<T, T>
```

### Usage

```typescript
import { merge, fromEvent } from "@cereb/core";

const click$ = fromEvent(document, "click");
const touch$ = fromEvent(document, "touchstart");

// Emit from both sources
const interaction$ = merge(click$, touch$);
```

### mergeWith (operator form)

```typescript
import { pipe, fromEvent } from "@cereb/core";
import { mergeWith } from "@cereb/core";

const combined$ = pipe(
  click$,
  mergeWith(touch$, keydown$)
);
```

## combineLatest

Emits array of latest values when any source emits.

```typescript
function combineLatest<T extends readonly Observable<unknown>[]>(
  ...sources: T
): Observable<{ [K in keyof T]: T[K] extends Observable<infer U> ? U : never }>
```

### Usage

```typescript
import { combineLatest } from "@cereb/core";

const combined$ = combineLatest(width$, height$);

combined$.subscribe(([width, height]) => {
  updateSize(width, height);
});
```

## forkJoin

Emits array of final values when all sources complete.

```typescript
function forkJoin<T extends readonly Observable<unknown>[]>(
  ...sources: T
): Observable<{ [K in keyof T]: T[K] extends Observable<infer U> ? U : never }>
```

### Usage

```typescript
import { forkJoin, fromPromise } from "@cereb/core";

const results$ = forkJoin(
  fromPromise(fetchUser()),
  fromPromise(fetchSettings()),
  fromPromise(fetchPermissions())
);

results$.subscribe(([user, settings, permissions]) => {
  initializeApp(user, settings, permissions);
});
```
